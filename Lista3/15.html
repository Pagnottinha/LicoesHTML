Par ou ímpar

Escreva um programa que joga par ou ímpar com o usuário.

            #include <stdio.h>
            #include <time.h>
            #include <stdlib.h>
            
            int main(void) {
            
                printf("Bora jogar par ou ímpar?\n");
            
                int decisao;
                printf("Digite 0 para par e 1 para ímpar: ");
                scanf("%d", &decisao);
            
                if (decisao == 0) {
                    printf("Você escolheu par!\n");
                }
                else
                {
                    printf("Você escolheu ímpar!\n");
                }
                
                int numDedosJogador;
                printf("Quantos dedos você colocou? ");
                scanf("%d", &numDedosJogador);
            
                
                srand(time(NULL));
                int numDedosMaquina = rand() % 5 + 1;
                int resultado = (numDedosMaquina + numDedosJogador) % 2;
            
                if (resultado == decisao) {
                    printf("Você ganhou!\nEu coloquei %d dedos\n", numDedosMaquina);
                }
                else
                {
                    printf("Você perdeu!\nEu coloquei %d dedos\n", numDedosMaquina);
                }
                
                return 0;
            }
